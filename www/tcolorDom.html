    Fun with toxi/color/TColor and the DOM with friend: jQuery

    require([
  'jquery',
  'toxi/color/TColor'
], function( $, TColor ){
  var $el = $("#example-container").height( window.innerHeight - 50 );
  var tbg = TColor.newCSS( $el.css('background-color') );

  console.log( tbg );
  tbg.setHue( 0.125 );
  tbg.setSaturation( 0.5 );

  $el.css('background-color', tbg.toRGBACSS() );
});
